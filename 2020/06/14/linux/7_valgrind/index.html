    Linux C++ valgrind 内存分析
    2020/06/14 学习 CPP Linux

    valgrind：用于分析 C++ 程序内存使用的异常情况，比如内存泄漏、越界访问、未初始化

    // main.cpp
#include <iostream>
#include <cstdlib>
#include <vector>

using namespace std;

void print_usage(char * argv) {
    cout << "usage: " << argv << " 1|2|3" << endl;
}

// 内存泄漏
void f1() {
    int *array = (int *)malloc(sizeof(int));
}

// 越界访问
void f2() {
    vector<int> v(10, 0);
    cout << v[10] << endl;
}

// 未初始化
void f3() {
    int x;
    if (x == 0) {
        std::cout << "X is zero" << std::endl;
    }
}

int main(int argc, char* argv[]) {
    if (argc < 2) {
        print_usage(argv[0]);
        return -1;
    }

    switch (atoi(argv[1])) {
        case 1:
            f1();
            break;
        case 2:
            f2();
            break;
        case 3:
            f3();
            break;
        default:
            print_usage(argv[0]);
            break;
    }
    
    return 0;
}
