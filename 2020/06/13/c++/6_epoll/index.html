    C++ epoll
    2020/06/13 学习 CPP
    1. epoll的优点
      • fd上限。支持的fd上限是最大可以打开文件的数目，这个数字一般远大于select的1024；
      • 事件通知。通过epoll_ctl注册fd，一旦该fd就绪，内核就会采用类似callback的回调机制来激活该fd，epoll_wait便可以收到通知，而非select、poll的轮询机制；
      • 减少内存拷贝。只在epoll_ctl时把current挂一遍（这一遍必不可少）并为每个fd指定一个回调函数，当设备就绪，唤醒等待队列上的等待者时，就会调用这个回调函数，而这个回调函数会把就绪的fd加入一个就绪链表）。epoll_wait的工作实际上就是在这个就绪链表中查看有没有就绪的fd（利用schedule_timeout()实现睡一会，判断一会的效果）。select或poll一样每次都把current轮流加入fd对应的设备等待队列中。
    2. epoll的实现
    // server.cpp
#include <sys/epoll.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/socket.h>
#include <cassert>
#include <fcntl.h>
#include <netinet/in.h>
#include <cstring>
#include <arpa/inet.h>
#include <unistd.h>
#include <iostream>

#define MAX_EVENTS 10

bool setnonblocking(int sd);
void do_use_fd(int fd);

int main() {
    struct epoll_event ev, events[MAX_EVENTS];
    int listen_sock, conn_sock, nfds, epollfd;

    /* Code to set up listening socket, 'listen_sock',
        (socket(), bind(), listen()) omitted */
    {
        const char* ip = "0.0.0.0";
        int port = 36363;
        sockaddr_in addr;
        memset(&addr, 0, sizeof(addr));
        addr.sin_family = AF_INET;
        addr.sin_port = htons(port);
        if(inet_aton(ip, &addr.sin_addr) == 0)
            return -1;

        // create socket
        if((listen_sock = socket(AF_INET, SOCK_STREAM, 0)) < 0)
            return -1;

        if(!setnonblocking(listen_sock))
            return -1;

        // bind socket
        if(bind(listen_sock, (sockaddr*)&addr, sizeof(addr)) != 0)
        {
            shutdown(listen_sock, 2);
	        close(listen_sock);
            return -1;
        }

        // listen
        if(listen(listen_sock, 10) != 0)
        {
            shutdown(listen_sock, 2);
	        close(listen_sock);
            return -1;
        }
    }

    epollfd = epoll_create1(0);
    if (epollfd == -1) {
        perror("epoll_create1");
        exit(EXIT_FAILURE);
    }

    ev.events = EPOLLIN;
    ev.data.fd = listen_sock;
    if (epoll_ctl(epollfd, EPOLL_CTL_ADD, listen_sock, &ev) == -1) {
        perror("epoll_ctl: listen_sock");
        exit(EXIT_FAILURE);
    }

    for (;;) {
        nfds = epoll_wait(epollfd, events, MAX_EVENTS, -1);
        if (nfds == -1) {
            perror("epoll_wait");
            exit(EXIT_FAILURE);
        }

        for (int n = 0; n < nfds; ++n) {
            if (events[n].data.fd == listen_sock) {
                struct sockaddr addr;
                socklen_t addrlen;
                conn_sock = accept(listen_sock,
                                    (struct sockaddr *) &addr, &addrlen);
                if (conn_sock == -1) {
                    perror("accept");
                    exit(EXIT_FAILURE);
                }
                setnonblocking(conn_sock);
                ev.events = EPOLLIN | EPOLLET;
                ev.data.fd = conn_sock;
                if (epoll_ctl(epollfd, EPOLL_CTL_ADD, conn_sock,
                            &ev) == -1) {
                    perror("epoll_ctl: conn_sock");
                    exit(EXIT_FAILURE);
                }
            } else {
                do_use_fd(events[n].data.fd);
            }
        }
    }
}

bool setnonblocking(int sd) {
    assert(sd >= 0);

    return fcntl(sd, F_SETFL, fcntl(sd, F_GETFD, 0) | O_NONBLOCK)  == 0;
}

void do_use_fd(int fd) {
    std::cout << "fd: " << fd << std::endl;

    shutdown(fd, 2);
    close(fd);
}

#include <arpa/inet.h>
#include <stdio.h>
#include <sys/socket.h>
#include <unistd.h>
#include <netdb.h>
#include <strings.h>

#ifndef ADDRESS
#define ADDRESS "127.0.0.1"
#endif

#ifndef PORT
#define PORT 36363
#endif

int main() {
	int                ret = 0;
	int                conn_fd;

    if ((conn_fd = socket(AF_INET, SOCK_STREAM, 0)) == -1) {
	    perror("Socket");
	    return -1;
	}

	struct sockaddr_in server_addr = { 0 };
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(PORT);

    ret = inet_pton(AF_INET, ADDRESS, &server_addr.sin_addr);
	if (ret != 1) {
		if (ret == -1) {
			perror("inet_pton");
		}
		fprintf(stderr,
		        "failed to convert address %s "
		        "to binary net address\n",
		        ADDRESS);
		return -1;
	}

	fprintf(stdout, "CONNECTING: address=%s port=%d\n", ADDRESS, PORT);

    ret =
	  connect(conn_fd, (struct sockaddr*)&server_addr, sizeof(server_addr));
	if (ret == -1) {
		perror("connect");
		return -1;
	}

    // After the connection has been properly established, we could go
	// forward with `read(2)` and `write(2)` calls.

    ret = shutdown(conn_fd, SHUT_RDWR);
	if (ret == -1) {
		perror("shutdown");
		return -1;
	}

    ret = close(conn_fd);
	if (ret == -1) {
		perror("close");
		return -1;
	}

	return 0;
}
